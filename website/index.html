Jnario

Executable Specifications for Java.

Describe your features in text...

...and make them executable.

‹ ›

Installation

Jnario requires Eclipse 3.6 with Xtext 2.3 and Xtend 2.3. You can install both via this Xtext update site.

Introduction

package example

import java.util.Stack
import java.util.EmptyStackException

describe Stack{
  context "empty stack"{
    fact subject.size should be 0
    fact subject.pop throws EmptyStackException
  }
  context "one element"{ 
    before subject.add("element")
    fact subject.size should be 1
    fact "pop decreases size"{
      subject.pop
      subject.size => 0 
    }
    fact "pop removes last element"{
      subject.pop => "element"
      subject.contains("element") => true
    }  
  }   
}
